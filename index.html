      function handleEmailFileSelect(evt){
      var zone=document.getElementById('emailDropZone');
      var file=evt.target.files&&evt.target.files[0];
      if(!file)return;
      evt.target.value='';
      var url=((document.getElementById('tEmailOrdner')||{}).value||'').trim()||((document.getElementById('tGDOrdner')||{}).value||'').trim();
      if(!url.trim()){_ezMsg(zone,'error','Bitte zuerst E-Mail-Ablageordner oder Projektordner in den Verknüpfungen eintragen!');return;}
      var fid=_ezFolderId(url);
      if(!fid){_ezMsg(zone,'error','Ungueltige Google Drive Ordner-URL.');return;}
      if(!gdriveToken){_ezMsg(zone,'error','Bitte zuerst mit Google Drive verbinden (oben).');return;}
      _ezMsg(zone,'','Verarbeite '+file.name+'...');
      // PDF direkt hochladen
      if(file.type==='application/pdf'||/\.pdf$/i.test(file.name)){
      var reader=new FileReader();
      reader.onload=function(e){
      _ezUpload(e.target.result,file.name,fid).then(function(){
      _ezMsg(zone,'success','✅ Gespeichert: '+file.name);
      setTimeout(function(){_ezMsg(zone,'','');},4000);
      }).catch(function(err){_ezMsg(zone,'error','Fehler: '+err.message);});
      };
      reader.readAsArrayBuffer(file);
      return;
      }
      var info={subject:file.name.replace(/\.(msg|eml)$/i,''),sender:'',date:new Date(),bodyText:''};
      // Try to read .eml as text
      if(file.name.toLowerCase().endsWith('.eml')){
      var reader=new FileReader();
      reader.onload=function(e){
      var parsed=_parseEmlText(e.target.result||'');
      if(!parsed.subject)parsed.subject=file.name.replace(/\.eml$/i,'');
      _ezProcess(parsed,fid,zone);
      };
      reader.readAsText(file);
      } else {
      _ezProcess(info,fid,zone);
      }
      }
      async function handleEmailDrop(dt,zone){
      var url=((document.getElementById('tEmailOrdner')||{}).value||'').trim()||((document.getElementById('tGDOrdner')||{}).value||'').trim();
      if(!url.trim()){_ezMsg(zone,'error','Bitte zuerst E-Mail-Ablageordner oder Projektordner in den Verknüpfungen eintragen!');return;}
      var fid=_ezFolderId(url);
      if(!fid){_ezMsg(zone,'error','Ungueltige Google Drive Ordner-URL.');return;}
      if(!gdriveToken){_ezMsg(zone,'error','⚠ Bitte zuerst mit Google Drive verbinden (oben rechts: ☁️ Google Drive).');return;}
      _ezMsg(zone,'','Verarbeite...');
      // 0. PDF direkt hochladen
      if(dt.files&&dt.files.length){
      var pdfs=[].slice.call(dt.files).filter(function(x){return x.type==='application/pdf'||/\.pdf$/i.test(x.name);});
      if(pdfs.length){
      var uploaded=[];
      for(var pi=0;pi<pdfs.length;pi++){
      var pf=pdfs[pi];
      try{
      var buf=await new Promise(function(res,rej){var r=new FileReader();r.onload=function(e){res(e.target.result);};r.onerror=rej;r.readAsArrayBuffer(pf);});
      await _ezUpload(buf,pf.name,fid);
      uploaded.push(pf.name);
      }catch(err){_ezMsg(zone,'error','Fehler bei '+pf.name+': '+err.message);return;}
      }
      _ezMsg(zone,'success','✅ Gespeichert: '+uploaded.join(', '));
      setTimeout(function(){_ezMsg(zone,'','');},4000);
      return;
      }
      }
      // Check what Outlook provides
      var types=[].slice.call(dt.types||[]);
      var info=null;
      // 1. File drop (.msg or .eml)
      if(dt.files&&dt.files.length){
      var f=[].slice.call(dt.files).find(function(x){return/\.(msg|eml)$/i.test(x.name);});
      if(f){
      if(f.name.toLowerCase().endsWith('.eml')){
      var text=await new Promise(function(res){var r=new FileReader();r.onload=function(e){res(e.target.result);};r.readAsText(f);});
      info=_parseEmlText(text);
      if(!info.subject)info.subject=f.name.replace(/\.(msg|eml)$/i,'');
      } else {
      info={subject:f.name.replace(/\.msg$/i,''),sender:'',date:new Date(),bodyText:''};
      }
      }
      }
      // 2. HTML from Outlook drag (direct from inbox)
      if(!info&&types.indexOf('text/html')>=0){
      var h=dt.getData('text/html');
      if(h&&h.length>20)info=_ezParseHtml(h);
      }
      // 3. Plain text fallback
      if(!info&&types.indexOf('text/plain')>=0){
      var t=dt.getData('text/plain');
      if(t&&t.trim()){
      var s=t.split('\n')[0].replace(/^(Betreff|Subject):\s*/i,'').trim()||'E-Mail';
      info={subject:s,sender:'',date:new Date(),bodyText:t};
      }
      }
      if(!info){
      _ezMsg(zone,'error','Keine E-Mail-Daten erkannt. Bitte .msg-Datei per Knopf auswaehlen.');
      return;
      }
      _ezProcess(info,fid,zone);
      }
      async function _ezProcess(info,fid,zone){
      try{
      var pdf=await _ezBuildPdfFromHtml(info);
      var d=info.date instanceof Date&&!isNaN(info.date)?info.date:new Date();
      var stamp=d.getFullYear()+'.'+_ez2(d.getMonth()+1)+'.'+_ez2(d.getDate());
      var fname=stamp+'_'+_ezClean(info.subject)+'.pdf';
      var result=await _ezUpload(pdf,fname,fid);
      _ezMsg(zone,'success','✅ Gespeichert: '+fname);
      setTimeout(function(){document.getElementById('emailDropZoneMsg')&&_ezMsg(zone,'','');},4000);
      // PDF-Vorschau im Task speichern und Sektion öffnen
      if(result&&result.id){
      if(!taskEmailPdfs.some(p=>p.fileId===result.id)){
      taskEmailPdfs.push({
      name:fname,
      fileId:result.id,
      url:'https://drive.google.com/file/d/'+result.id+'/view',
      uploadedAt:new Date().toLocaleDateString('de-DE')
      });
      }
      if(typeof renderEmailSection==='function')renderEmailSection();
      if(typeof scheduleTaskFieldSave==='function')scheduleTaskFieldSave();
      // E-Mail-Sektion öffnen und zur Vorschau scrollen
      var sect=document.getElementById('tEmailOrigSect');
      if(sect&&!sect.classList.contains('open'))sect.classList.add('open');
      setTimeout(function(){
      var grid=document.getElementById('emailIconGrid');
      if(grid)grid.scrollIntoView({behavior:'smooth',block:'start'});
      },200);
      const allBoards=Object.keys(boards).filter(b=>!b.startsWith('_')&&b!==currentBoard);
      if(!allBoards.length){showToast('Keine anderen Boards vorhanden.','info');return;}
      const old=document.getElementById('_moveBoardPop');if(old)old.remove();
      const pop=document.createElement('div');
      pop.id='_moveBoardPop';

      let html='<div style="font-size:14px;font-weight:700;color:var(--text);margin-bottom:14px;">↗ Task verschieben nach…</div>';
      html+='<div style="display:flex;flex-direction:column;gap:6px;">';
      for(const b of allBoards){
      html+=`<button onclick="event.stopPropagation();_doMoveTask('${esc(id)}','${esc(b)}')" style="text-align:left;padding:10px 14px;border-radius:8px;border:1px solid var(--border);background:var(--bg3);color:var(--text);cursor:pointer;font-size:13px;">${esc(b)}</button>`;
      }

      html+='<button onclick="document.getElementById(\'_moveBoardPop\').remove()" style="margin-top:14px;width:100%;padding:8px;border-radius:8px;border:1px solid var(--border);background:transparent;color:var(--text2);cursor:pointer;font-size:12px;">Abbrechen</button>';
      pop.innerHTML=html;

      setTimeout(()=>document.addEventListener('click',function _cl(e){if(!pop.contains(e.target)){pop.remove();document.removeEventListener('click',_cl);}},{ once: false }),50);
      }
      function _doMoveTask(taskId,targetBoard){
      const old=document.getElementById('_moveBoardPop');if(old)old.remove();
      let task=null,srcBoard=null;
      for(const b of Object.keys(boards)){
      const idx=(boards[b]||[]).findIndex(t=>t.id===taskId);
      if(idx>=0){task=boards[b].splice(idx,1)[0];srcBoard=b;break;}
      }
      if(!task){showToast('Task nicht gefunden.','error');return;}
      const targetStatuses=boards['_statuses']&&boards['_statuses'][targetBoard]
      ?boards['_statuses'][targetBoard]
      :['Backlog'];
      task.status=targetStatuses[0]||'Backlog';
      if(!boards[targetBoard])boards[targetBoard]=[];
      boards[targetBoard].push(task);
      scheduleTaskFieldSave();
      closeModal();
      render();
      showToast('✅ Task nach „'+targetBoard+'" verschoben','success',3500);
      }
